fdisk /dev/sda
    g
    n +512M     # for /efi
    n +8192M    # for swap
    n +remain   # for /
    t 1 1       # for EFI System
    t 2 19      # for Linux swap
    t 3 20      # for Linux filesystem
    w           # write
